3. Insert Logic (Idempotent / Safe)
from app import db
from models import Resort

def insert_resort(name, state, country, pass_brands, slug):
    exists = Resort.query.filter_by(
        name=name,
        state=state,
        country=country
    ).first()

    if not exists:
        db.session.add(
            Resort(
                name=name,
                state=state,
                state_full=state,
                country=country,
                pass_brands=pass_brands,
                slug=slug,
                is_active=True
            )
        )
