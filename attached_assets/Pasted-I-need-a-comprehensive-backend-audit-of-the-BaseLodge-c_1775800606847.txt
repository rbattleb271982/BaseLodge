============================================================

Run these SQL queries against the production database and paste

Q1. Total users:
    SELECT COUNT(*) FROM "user";

Q2. Total SkiTrip records:
    SELECT COUNT(*) FROM ski_trip;

Q3. Total SkiTripParticipant rows:
    SELECT COUNT(*) FROM ski_trip_participant;

Q4. Distinct SkiTrips that have at least one SkiTripParticipant:
    SELECT COUNT(DISTINCT trip_id) FROM ski_trip_participant;

Q5. SkiTrips with more than one SkiTripParticipant 
    (actually shared trips):
    SELECT COUNT(*) FROM (
      SELECT trip_id FROM ski_trip_participant 
      GROUP BY trip_id HAVING COUNT(*) > 1
    ) AS shared;

Q6. SkiTrips whose owner does NOT have a corresponding 
    SkiTripParticipant row with role=OWNER:
    SELECT COUNT(*) FROM ski_trip st
    WHERE NOT EXISTS (
      SELECT 1 FROM ski_trip_participant stp
      WHERE stp.trip_id = st.id 
        AND stp.user_id = st.user_id 
        AND stp.role = 'OWNER'
    );
    Tell me what the actual stored value for the OWNER role is 
    if it differs from 'OWNER'.

Q7. SkiTrips with more than one OWNER participant row 
    (would violate "exactly one owner" constraint):
    SELECT COUNT(*) FROM (
      SELECT trip_id FROM ski_trip_participant 
      WHERE role = 'OWNER'
      GROUP BY trip_id HAVING COUNT(*) > 1
    ) AS multi_owner;

Q8. Duplicate (trip_id, user_id) participant rows 
    (would violate "one row per user per trip" constraint):
    SELECT COUNT(*) FROM (
      SELECT trip_id, user_id FROM ski_trip_participant 
      GROUP BY trip_id, user_id HAVING COUNT(*) > 1
    ) AS dupes;

Q9. Total GroupTrip records:
    SELECT COUNT(*) FROM group_trip;

Q10. Total TripGuest records:
     SELECT COUNT(*) FROM trip_guest;

Q11. Total Friend rows:
     SELECT COUNT(*) FROM friend;

Q12. Number of mutual friendships (both directions exist):
     SELECT COUNT(*) FROM friend f1
     WHERE EXISTS (
       SELECT 1 FROM friend f2
       WHERE f2.user_id = f1.friend_id 
         AND f2.friend_id = f1.user_id
     );

Q13. Number of one-way friendships (only one direction exists):
     SELECT COUNT(*) FROM friend f1
     WHERE NOT EXISTS (
       SELECT 1 FROM friend f2
       WHERE f2.user_id = f1.friend_id 
         AND f2.friend_id = f1.user_id
     );

Q14. Total Invitation records, grouped by type and status:
     SELECT invite_type, status, COUNT(*) 
     FROM invitation 
     GROUP BY invite_type, status;

Q15. Total Event records (if the Event table exists):
     SELECT COUNT(*) FROM event;
     If it exists, also run:
     SELECT event_name, COUNT(*) FROM event GROUP BY event_name;

Q16. Total Activity records:
     SELECT COUNT(*) FROM activity;
     If it exists, also run:
     SELECT activity_type, COUNT(*) FROM activity 
     GROUP BY activity_type;
