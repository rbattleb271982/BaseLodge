Display a logged-in user’s ski trips on a mobile-first page, split into “Upcoming” and “Past” sections. Include resort name, dates, pass, and whether it’s public/private. Add buttons to edit or delete trips.
1. 🧠 routes/user.py — Add My Trips route
from datetime import date

@user_bp.route('/my-trips')
@login_required
def my_trips():
    today = date.today()

    upcoming_trips = Trip.query.filter(
        Trip.user_id == current_user.id,
        Trip.end_date >= today
    ).order_by(Trip.start_date).all()

    past_trips = Trip.query.filter(
        Trip.user_id == current_user.id,
        Trip.end_date < today
    ).order_by(Trip.start_date.desc()).all()

    return render_template(
        'user/my_trips.html',
        upcoming_trips=upcoming_trips,
        past_trips=past_trips
    )

def delete_trip(trip_id):
    trip = Trip.query.get_or_404(trip_id)
    if trip.user_id != current_user.id:
        flash("Not authorized")
        return redirect(url_for('user.my_trips'))

    db.session.delete(trip)
    db.session.commit()
    flash("Trip deleted")
    return redirect(url_for('user.my_trips'))
