def pass_category(pass_type):
    if pass_type in ["Epic", "Epic Local", "Epic Pass"]:
        return "Epic"
    if pass_type in ["Ikon", "Ikon Base"]:
        return "Ikon"
    return "Other"

@app.route("/friends")
@login_required
def friends():
    filter_type = request.args.get("filter", "All")

    # Load friend relationships
    friend_links = Friend.query.filter_by(user_id=current_user.id).all()
    friend_ids = [f.friend_id for f in friend_links]
    friends = User.query.filter(User.id.in_(friend_ids)).all()

    # Categorize friends
    epic_friends = [f for f in friends if pass_category(f.pass_type) == "Epic"]
    ikon_friends = [f for f in friends if pass_category(f.pass_type) == "Ikon"]
    other_friends = [f for f in friends if pass_category(f.pass_type) == "Other"]

    if filter_type == "Epic":
        friends = epic_friends
    elif filter_type == "Ikon":
        friends = ikon_friends
    elif filter_type == "Other":
        friends = other_friends

    return render_template(
        "friends.html",
        friends=friends,
        count_epic=len(epic_friends),
        count_ikon=len(ikon_friends),
        count_other=len(other_friends),
        filter_type=filter_type
    )
