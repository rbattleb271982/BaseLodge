3. CREATE NEW ROUTE: /select-pass
------------------------------------------------------------

In app.py, add:

@app.route("/select-pass", methods=["GET", "POST"])
@login_required
def select_pass():
    # Curated pass list for now
    major_passes = ["Epic", "Ikon", "Indy", "Mountain Collective"]
    regional_passes = ["Power Pass", "Boyne Pass", "A-Basin Pass", "Loveland Pass"]
    other_passes = ["Other", "None"]

    if request.method == "POST":
        chosen = request.form.get("pass_type")
        current_user.pass_type = chosen
        db.session.commit()
        session["pass_prompt_skipped"] = False  # show card again only if no pass selected in future
        return redirect(url_for("home"))

    return render_template(
        "select_pass.html",
        major=major_passes,
        regional=regional_passes,
        other=other_passes
    )

<script>
const items = document.querySelectorAll(".pass-item");
const hiddenInput = document.getElementById("pass_type");
const saveBtn = document.getElementById("saveBtn");

items.forEach(item => {
    item.addEventListener("click", () => {
        items.forEach(i => i.classList.remove("selected"));
        item.classList.add("selected");
        hiddenInput.value = item.dataset.value;
        saveBtn.disabled = false;
    });
});

// Search filter
document.getElementById("search").addEventListener("input", function() {
    const text = this.value.toLowerCase();
    items.forEach(item => {
        item.style.display = item.textContent.toLowerCase().includes(text) ? "flex" : "none";
    });
});
</script>

</body>
