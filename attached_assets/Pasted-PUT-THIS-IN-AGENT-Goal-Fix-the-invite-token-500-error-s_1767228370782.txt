@app.template_filter('identity_line')
def identity_line_filter(user):
    try:
        if not user:
            return ""

        parts = []

        if getattr(user, "rider_type", None):
            parts.append(str(user.rider_type))

        passes = getattr(user, "pass_brands", None)
        if passes:
            if isinstance(passes, list):
                parts.append(", ".join([str(p) for p in passes]))
            else:
                parts.append(str(passes))

        if getattr(user, "skill_level", None):
            parts.append(str(user.skill_level))

        return " · ".join(parts)

    except Exception:
        return ""
