HTML — templates/home.html (current user)
Replace the identity section with:
<div class="profile-identity">
  <div class="profile-identity-name">
    {{ user.first_name }}
  </div>

  <div class="profile-identity-pills">
    {% if user.rider_type %}
      <span class="profile-pill">{{ normalize_rider_type(user.rider_type) }}</span>
    {% endif %}
    {% if user.pass_type %}
      {% set passes = user.pass_type.split(',') %}
      {% for p in passes %}
        <span class="profile-pill">{{ p }}</span>
      {% endfor %}
    {% endif %}
    {% if user.skill_level %}
      <span class="profile-pill">{{ user.skill_level }}</span>
    {% endif %}
  </div>

  <div class="profile-identity-rows">
    {% if user.terrain_preferences and user.terrain_preferences|length > 0 %}
      <div class="profile-identity-row">
        <span class="profile-identity-label">Preferred Terrain</span>
        <span class="profile-identity-sep">·</span>
        <span class="profile-identity-value">
          {{ user.terrain_preferences[:2]|join(', ') }}
        </span>
      </div>
    {% endif %}

    {% if eq and eq.brand %}
      <div class="profile-identity-row">
        <span class="profile-identity-label">Equipment</span>
        <span class="profile-identity-sep">·</span>
        <span class="profile-identity-value">
          {{ eq.brand }}{% if eq.model %} {{ eq.model }}{% endif %}
        </span>
      </div>
    {% endif %}
  </div>
</div>
