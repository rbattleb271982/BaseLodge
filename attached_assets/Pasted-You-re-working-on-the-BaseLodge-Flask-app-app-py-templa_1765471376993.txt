Add a small <script> at the bottom of mountains_visited.html to:
Show only the .mountain-state-group whose data-state matches the selected state.
Filter visible labels by search text (case-insensitive) within that state.
Example:
<script>
  const stateSelect = document.getElementById("state-select");
  const searchInput = document.getElementById("mountain-search");
  const groups = document.querySelectorAll(".mountain-state-group");

  function updateVisibleMountains() {
    const state = stateSelect.value;
    const query = searchInput.value.toLowerCase();

    groups.forEach(group => {
      if (!state || group.dataset.state !== state) {
        group.style.display = "none";
        return;
      }
      group.style.display = "block";

      const options = group.querySelectorAll(".mountain-option");
      options.forEach(opt => {
        const name = opt.dataset.mountainName || "";
        opt.style.display = name.includes(query) ? "flex" : "none";
      });
    });
  }

  stateSelect.addEventListener("change", () => {
    searchInput.value = "";
    updateVisibleMountains();
  });

  searchInput.addEventListener("input", updateVisibleMountains);

  // Initialize (hide all until state selected)
  updateVisibleMountains();
</script>
