1. 📦 models.py — Add Trip model
from your_app import db
from flask_login import UserMixin

class Trip(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)
    resort_name = db.Column(db.String(100), nullable=False)
    start_date = db.Column(db.Date, nullable=False)
    end_date = db.Column(db.Date, nullable=False)
    pass_type = db.Column(db.String(50), nullable=False)
    is_private = db.Column(db.Boolean, default=False)

    user = db.relationship('User', backref='trips')

2. 🧠 routes/user.py — Add Routes
from flask import render_template, request, redirect, url_for, flash
from flask_login import login_required, current_user
from datetime import datetime
from models import Trip, db

@user_bp.route('/add-trip', methods=['GET', 'POST'])
@login_required
def add_trip():
    if request.method == 'POST':
        resort_name = request.form['resort']
        start_date = datetime.strptime(request.form['start_date'], '%Y-%m-%d').date()
        end_date = datetime.strptime(request.form['end_date'], '%Y-%m-%d').date()
        pass_type = request.form['pass_type']
        is_private = True if request.form.get('is_private') == 'on' else False

        # Prevent overlapping trips
        overlapping = Trip.query.filter(
            Trip.user_id == current_user.id,
            Trip.start_date <= end_date,
            Trip.end_date >= start_date
        ).first()

        if overlapping:
            flash("You already have a trip that overlaps with these dates.")
            return redirect(url_for('user.add_trip'))

        new_trip = Trip(
            user_id=current_user.id,
            resort_name=resort_name,
            start_date=start_date,
            end_date=end_date,
            pass_type=pass_type,
            is_private=is_private
        )
        db.session.add(new_trip)
        db.session.commit()
        return redirect(url_for('user.my_trips'))

    return render_template('user/add_trip.html')
