document.getElementById('invite-friend-btn').addEventListener('click', function() {
    const message = "Hey! I’m using BaseLodge to connect with friends for ski trips. You should join — it’s free and makes planning super easy: https://baselodge.app";
    
    if (navigator.share) {
        navigator.share({ text: message });
    } else {
        window.location.href = "sms:?body=" + encodeURIComponent(message);
    }
});
</script>
4️⃣ FRIEND FILTERS ON FRIENDS PAGE (friends.html)
Add filter pills for pass types:
Epic | Ikon | Other | None
HTML
<div class="friend-filter-bar">
  <button class="filter-pill" data-filter="Epic">Epic</button>
  <button class="filter-pill" data-filter="Ikon">Ikon</button>
  <button class="filter-pill" data-filter="Other">Other</button>
  <button class="filter-pill" data-filter="None">None</button>
</div>
JS Logic
<script>
const pills = document.querySelectorAll('.filter-pill');

pills.forEach(pill => {
  pill.addEventListener('click', () => {
    const pass = pill.dataset.filter;

    document.querySelectorAll('.friend-card').forEach(card => {
      if (pass === 'None' || card.dataset.pass === pass) {
        card.style.display = 'block';
      } else {
        card.style.display = 'none';
      }
    });
  });
});
</script>
CSS
.friend-filter-bar {
    display: flex;
    gap: 8px;
    margin-bottom: 16px;
}

.filter-pill {
    padding: 8px 12px;
    background: #F7F7F7;
    border: 1px solid #DDD;
    border-radius: 20px;
    font-size: 0.9rem;
}
