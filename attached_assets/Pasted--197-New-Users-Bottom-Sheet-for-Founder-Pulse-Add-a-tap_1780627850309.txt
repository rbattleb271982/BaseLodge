#197 - New Users Bottom Sheet for Founder Pulse

Add a tappable New Users drilldown to the Founder Pulse section on the admin dashboard.

Goal:
When I tap the “NEW USERS” Founder Pulse card, open a native-style bottom sheet listing users who signed up today.

Files:
- app.py
- templates/admin_dashboard.html

Backend:
In app.py, insert this new endpoint after admin_api_active_today_users:

@app.route("/admin/api/new-users-today")
@login_required
@admin_required
def admin_api_new_users_today():
    """JSON list of users who signed up today — same Denver midnight as Founder Pulse."""
    today_start = _admin_today_start_utc()

    users = User.query.filter(User.created_at >= today_start)\
                      .order_by(User.created_at.desc()).all()

    result = []
    for u in users:
        result.append({
            "user_id":    u.id,
            "first_name": (u.first_name or "").strip(),
            "last_name":  (u.last_name  or "").strip(),
            "email":      u.email or "",
            "created_at": u.created_at.isoformat() if u.created_at else None,
        })

    return jsonify(result)

Requirements:
- Use _admin_today_start_utc()
- Do not change Founder Pulse KPI calculations
- Do not change the Active Today endpoint

4. Add this JS after the existing Active Today sheet JS:

(function () {
  var overlay  = document.getElementById('fp-nu-overlay');
  var sheet    = document.getElementById('fp-nu-sheet');
  var titleEl  = document.getElementById('fp-nu-title');
  var body     = document.getElementById('fp-nu-body');
  var closeBtn = document.getElementById('fp-nu-close');

  function fmtSignupTime(isoStr) {
    if (!isoStr) return '';
    var d = new Date(isoStr.endsWith('Z') ? isoStr : isoStr + 'Z');
    return d.toLocaleTimeString('en-US', {
      hour: 'numeric',
      minute: '2-digit',
      hour12: true,
      timeZone: 'America/Denver'
    });
  }

  function escHtml(s) {
    return String(s || '')
      .replace(/&/g,'&amp;')
      .replace(/</g,'&lt;')
      .replace(/>/g,'&gt;')
      .replace(/"/g,'&quot;');
  }

  function openNuSheet(users) {
    var n = users.length;
    titleEl.textContent = n === 1 ? '1 New User Today' : n + ' New Users Today';

    if (n === 0) {
      body.innerHTML = '<div class="fp-sheet-empty">No new users today.</div>';
    } else {
      body.innerHTML = users.map(function (u) {
        var first = escHtml(u.first_name);
        var last  = escHtml(u.last_name);
        var name  = (first + ' ' + last).trim() || 'Unknown User';
        var email = escHtml(u.email);
        var time  = escHtml(fmtSignupTime(u.created_at));

        return '<a class="fp-sheet-row" href="/admin/users/' + u.user_id + '">' +
          '<span class="fp-sheet-row-name" style="display:flex;flex-direction:column;gap:2px;">' +
            '<span>' + name + '</span>' +
            '<span style="font-size:12px;color:#9A9186;font-weight:400;">' + email + '</span>' +
          '</span>' +
          '<span class="fp-sheet-row-time">' + time + '</span>' +
          '</a>';
      }).join('');
    }

    overlay.classList.add('fp-open');
    sheet.classList.add('fp-open');
    document.body.style.overflow = 'hidden';
  }

  function closeNuSheet() {
    overlay.classList.remove('fp-open');
    sheet.classList.remove('fp-open');
    document.body.style.overflow = '';
  }

  document.querySelectorAll('.fp-new-users-trigger').forEach(function (el) {
    el.addEventListener('click', function () {
      body.innerHTML = '<div class="fp-sheet-empty">Loading…</div>';
      overlay.classList.add('fp-open');
      sheet.classList.add('fp-open');
      document.body.style.overflow = 'hidden';

      fetch('/admin/api/new-users-today')
        .then(function (r) { return r.json(); })
        .then(openNuSheet)
        .catch(function () {
          body.innerHTML = '<div class="fp-sheet-empty">Unable to load new users.</div>';
        });
    });
  });

  closeBtn.addEventListener('click', closeNuSheet);
  overlay.addEventListener('click', closeNuSheet);
  document.addEventListener('keydown', function (e) {
    if (e.key === 'Escape') closeNuSheet();
  });
})();
