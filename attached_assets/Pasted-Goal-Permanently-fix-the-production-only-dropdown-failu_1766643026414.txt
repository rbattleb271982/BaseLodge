function safeSetSelect(select, value) {
  if (!value) {
    select.value = "";
    return;
  }

  const exists = [...select.options].some(o => o.value === value);
  if (exists) {
    select.value = value;
  } else {
    console.warn("Invalid dropdown value ignored:", value);
    select.value = "";
  }
}
