### 3) Implement /add_trip (GET + POST)

@app.route("/add_trip", methods=["GET", "POST"])
@login_required
def add_trip():
    if request.method == "POST":
        state = request.form.get("state")
        mountain = request.form.get("mountain")
        start_date_str = request.form.get("start_date")
        end_date_str = request.form.get("end_date")
        is_public = request.form.get("is_public") == "on"

        errors = []

        if not state:
            errors.append("Please select a state.")
        if not mountain:
            errors.append("Please select a mountain.")
        if not start_date_str:
            errors.append("Please select a start date.")
        if not end_date_str:
            errors.append("Please select an end date.")

        start_date = None
        end_date = None
        if start_date_str and end_date_str:
            try:
                start_date = datetime.strptime(start_date_str, "%Y-%m-%d").date()
                end_date = datetime.strptime(end_date_str, "%Y-%m-%d").date()
                if end_date < start_date:
                    errors.append("End date cannot be before start date.")
            except ValueError:
                errors.append("Invalid date format.")

        if errors:
            for e in errors:
                flash(e, "error")
            # Re-render the form with what we can
            return render_template(
                "add_trip.html",
                trip=None,
                form_action=url_for("add_trip"),
            )

        trip = SkiTrip(
            user_id=current_user.id,
            state=state,
            mountain=mountain,
            start_date=start_date,
            end_date=end_date,
            is_public=is_public,
        )
        db.session.add(trip)
        db.session.commit()
        flash("Trip added.", "success")
        return redirect(url_for("my_trips"))

    # GET
    return render_template(
        "add_trip.html",
        trip=None,
        form_action=url_for("add_trip"),
    )

@app.route("/my-trips")
@login_required
def my_trips():
    today = date.today()

    upcoming_trips = (
        SkiTrip.query
        .filter_by(user_id=current_user.id)
        .filter(SkiTrip.end_date >= today)
        .order_by(SkiTrip.start_date.asc())
        .all()
    )

    past_trips = (
        SkiTrip.query
        .filter_by(user_id=current_user.id)
        .filter(SkiTrip.end_date < today)
        .order_by(SkiTrip.start_date.desc())
        .all()
    )

    return render_template(
        "my_trips.html",
        upcoming_trips=upcoming_trips,
        past_trips=past_trips,
    )

<script>
  document.addEventListener("DOMContentLoaded", function () {
    const stateSelect = document.getElementById("state");
    const mountainSelect = document.getElementById("mountain");

    function loadMountainsForState(stateCode, currentMountain) {
      mountainSelect.innerHTML = "";
      mountainSelect.disabled = true;

      if (!stateCode) {
        const opt = document.createElement("option");
        opt.value = "";
        opt.textContent = "Select a state first";
        mountainSelect.appendChild(opt);
        return;
      }

      fetch(`/api/mountains/${stateCode}`)
        .then((res) => res.json())
        .then((data) => {
          mountainSelect.disabled = false;

          if (!Array.isArray(data) || data.length === 0) {
            const opt = document.createElement("option");
            opt.value = "";
            opt.textContent = "No mountains defined for this state yet";
            mountainSelect.appendChild(opt);
            mountainSelect.disabled = true;
            return;
          }

          const placeholder = document.createElement("option");
          placeholder.value = "";
          placeholder.textContent = "Select a mountain";
          mountainSelect.appendChild(placeholder);

          data.forEach((name) => {
            const opt = document.createElement("option");
            opt.value = name;
            opt.textContent = name;
            mountainSelect.appendChild(opt);
          });

          if (currentMountain) {
            mountainSelect.value = currentMountain;
          }
        })
        .catch(() => {
          const opt = document.createElement("option");
          opt.value = "";
          opt.textContent = "Error loading mountains";
          mountainSelect.appendChild(opt);
          mountainSelect.disabled = true;
        });
    }

    stateSelect.addEventListener("change", function () {
      loadMountainsForState(stateSelect.value, null);
    });

    // If we’re editing a trip, auto-load mountains for the selected state
    const existingState = stateSelect.value;
    const currentMountain = mountainSelect.dataset.currentMountain;
    if (existingState) {
      loadMountainsForState(existingState, currentMountain);
    }
  });
</script>
{% endblock %}
